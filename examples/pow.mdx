    result = 1.0

    return result
end

def near1pow2(n)
    if n == 0
        return 1.0
    elsif n == 1
        return 1.0000000001
    elsif n % 2 > 0
        return near1pow2(n-1) * 1.0000000001
    else
        return near1pow2(n/2)**2
    end
end
p near1pow2 2
`,  `
def near1pow3(n)
    return Math.exp(n*Math.log(1.0000000001))
end
